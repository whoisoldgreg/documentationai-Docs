## Quick Start

Get up and running with Scriptmasters in minutes.

<Steps>
  <Step title="Create Account" icon="user-plus">
    Sign up at `https://dashboard.scriptmasters.com` and verify your email.
  </Step>
  <Step title="Get API Key" icon="key">
    Navigate to the developer section to generate your `{API_KEY}`.
  </Step>
  <Step title="Make First Payment" icon="zap">
    Use this example to process a test transaction:

    <CodeGroup tabs="JavaScript,cURL">
      ```javascript
      const response = await fetch('https://api.example.com/v1/payments', {
        method: 'POST',
        headers: {
          'Authorization': `Bearer ${YOUR_API_KEY}`,
          'Content-Type': 'application/json'
        },
        body: JSON.stringify({
          amount: 1000,
          currency: 'USD',
          card: {
            number: '4111111111111111',
            exp_month: 12,
            exp_year: 2025,
            cvc: '123'
          }
        })
      });
      const data = await response.json();
      console.log(data);
      ```
      ```bash
      curl -X POST https://api.example.com/v1/payments \
        -H "Authorization: Bearer YOUR_API_KEY" \
        -H "Content-Type: application/json" \
        -d '{
          "amount": 1000,
          "currency": "USD",
          "card": {
            "number": "4111111111111111",
            "exp_month": 12,
            "exp_year": 2025,
            "cvc": "123"
          }
        }'
      ```
    </CodeGroup>
  </Step>
</Steps>
